14def set_axes_equal(ax_func):
15 """ Make axes of 3D plot have equal scale so that spheres appear as spheres,
16 cubes as cubes, etc.. This is one possible solution to Matplotlib's
17 ax.set_aspect('equal') and ax.axis('equal') not working for 3D.
18
19 Args:
20 ax_func (_type_): a matplotlib axis, e.g., as output from plt.gca().
21 """
22 x_limits = ax_func.get_xlim3d()
23 y_limits = ax_func.get_ylim3d()
24 z_limits = ax_func.get_zlim3d()
25
26 x_range = abs(x_limits[1] - x_limits[0])
27 x_middle = np.mean(x_limits)
28 y_range = abs(y_limits[1] - y_limits[0])
29 y_middle = np.mean(y_limits)
30 z_range = abs(z_limits[1] - z_limits[0])
31 z_middle = np.mean(z_limits)
32
33 plot_radius = 0.5*max([x_range, y_range, z_range])
34
35 ax_func.set_xlim3d([x_middle - plot_radius, x_middle + plot_radius])
36 ax_func.set_ylim3d([y_middle - plot_radius, y_middle + plot_radius])
37 ax_func.set_zlim3d([z_middle - plot_radius, z_middle + plot_radius])

39def data_for_cylinder_along_z(center_x,center_y,radius,height_z):
40 """ Creates set of coordinates (X, Y, Z) to draw a cylinder
41
42 Args:
43 center_x (double): the X coordinate center of the cylinder
44 center_y (double): the Y coordinate center of the cylinder
45 radius (double): the radius of the cylinder
46 height_z (double): the full height of the cylinder
47
48 Returns:
49 x_grid, y_grid, z_grid (ndarray) : set of coordinates of the
50 cylinder with given properties
51 """
52 z_vec = np.linspace(0, height_z, 50)-height_z/2
53 theta = np.linspace(0, 2*np.pi, 50)
54 theta_grid, z_grid=np.meshgrid(theta, z_vec)
55 x_grid = radius*np.cos(theta_grid) + center_x
56 y_grid = radius*np.sin(theta_grid) + center_y
57 return x_grid,y_grid,z_grid
